DECLARE
ANIO NUMBER:= 2005;
ES_BISIESTO BOOLEAN;
BEGIN
IF(MOD(ANIO,4)=0 AND MOD(ANIO,100)!=0 OR MOD(ANIO, 400)=0)THEN
ES_BISIESTO:=TRUE;
ELSE
ES_BISIESTO:=FALSE;
END IF;
IF(ES_BISIESTO)THEN
DBMS_OUTPUT.PUT_LINE(ANIO || ' es un año bisiesto.');
    ELSE
        DBMS_OUTPUT.PUT_LINE(ANIO || ' no es un año bisiesto.');
    END IF;
END;
