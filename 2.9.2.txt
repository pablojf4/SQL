CREATE TABLE ESPECTACULOS(
    COD_ESPECTACULO INT PRIMARY KEY,
    NOMBRE VARCHAR2(25),
    TIPO VARCHAR2(25),
    FECHA_INICIAL DATE,
    FECHA_FINAL DATE,
    INTERPRETE VARCHAR2(25),
    COD_RECINTO INT,
    CONSTRAINT COD_REC_FK
    REFERENCES RECINTOS(COD_RECINTO));

CREATE TABLE PRECIOS_ESPECTACULOS(
    COD_ESPECTACULO INT,
    COD_RECINTO INT,
    ZONA VARCHAR2(25),
    PRECIO DOUBLE,
    CONSTRAINT PRECIOS_ESPEC_PK PRIMARY KEY(COD_ESPECTACULO, COD_RECINTO, ZONA),
    CONSTRAINT PRE_ESP_FK FOREIGN KEY(COD_ESPECTACULO) REFERENCES ESPECTACULOS,
    CONSTRAINT PRE_ESP_FK FOREIGN KEY(COD_RECINTO,ZONA) REFERENCES ZONAS_RECINTOS(COD_RECINTO, ZONA));
CREATE TABLE RECINTOS(
    COD_RECINTO INT PRIMARY KEY,
    NOMBRE VARCHAR2(25),
    DIRECCION VARCHAR2(25),
    CIUDAD VARCHAR2(25),
    TELEFONO INT,
    HORARIO VARCHAR2(25));
CREATE TABLE ZONAS_RECINTOS(
    COD_RECINTO INT,
    CONSTRAINT ZON_REC_FK 
    REFERENCES RECINTOS,
    ZONA VARCHAR2(25),
    CAPACIDAD INT,
    CONSTRAINT ZON_REC_FK PRIMARY KEY(COD_RECINTO, ZONA));
CREATE TABLE ASIENTOS(
    COD_RECINTO INT,
    ZONA VARCHAR2(25),
    FILA INT,
    NUMERO INT,
    CONSTRAINT ASIEN_PK PRIMARY KEY(COD_RECINTO, ZONA, FILA, NUMERO),
    CONSTRAINT ASIEN_FK FOREIGN KEY(COD_RECINTO) REFERENCES RECINTOS(COD_RECINTO),
    CONSTRAINT ASIEN_FK FOREIGN KEY(ZONA) REFERENCES ZONAS_RECINTOS(ZONA));
CREATE TABLE REPRESENTACIONES(
    COD_ESPECTACULO INT,
    FECHA DATE,
    HORA TIMESTAMP
    CONSTRAINT REPRE_PK PRIMARY KEY(COD_ESPECTACULO, FECHA, HORA),
    CONSTRAINT REPRE_FK FOREIGN KEY(COD_ESPECTACULO) REFERENCES ESPECTACULOS);
